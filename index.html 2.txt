<script>

const PASSWORD = "1234";

function enviar() {
  let dados = JSON.parse(localStorage.getItem("fb")) || [];

  dados.push({
    nome: nome.value,
    nota: Number(nota.value),
    fortes: fortes.value,
    melhorar: melhorar.value
  });

  localStorage.setItem("fb", JSON.stringify(dados));

  alert("Enviado!");
}

function login() {
  if (pass.value !== PASSWORD) {
    alert("Errado!");
    return;
  }

  admin.style.display = "block";
  mostrar();
}

function mostrar() {
  let dados = JSON.parse(localStorage.getItem("fb")) || [];
  lista.innerHTML = "";

  let soma = 0;

  dados.forEach(d => {
    soma += d.nota;

    lista.innerHTML += `
      <div class="feedback-item">
        <b>${d.nome}</b><br>
        Nota: ${d.nota}<br>
        ${d.fortes}<br>
        ${d.melhorar}
      </div>
    `;
  });

  let media = dados.length ? (soma / dados.length).toFixed(2) : 0;
  media.innerHTML = "Média: " + media;

  grafico(dados);
}

function apagar() {
  localStorage.removeItem("fb");
  location.reload();
}

function grafico(dados) {
  let notas = [0,0,0,0,0];

  dados.forEach(d => notas[5-d.nota]++);

  new Chart(document.getElementById("grafico"), {
    type: "bar",
    data: {
      labels: ["5","4","3","2","1"],
      datasets: [{
        label: "Notas",
        data: notas,
        backgroundColor: "#667eea"
      }]
    }
  });
}

</script>
